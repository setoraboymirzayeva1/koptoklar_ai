async function init() { const modelURL = URL + "model.json"; const metadataURL = URL + "metadata.json"; model = await tmImage.load(modelURL, metadataURL); maxPredictions = model.getTotalClasses(); webcam = new tmImage.Webcam(350, 350, true); await webcam.setup(); await webcam.play(); window.requestAnimationFrame(loop); document.getElementById("webcam-container").appendChild(webcam.canvas); labelContainer = document.getElementById("label-text"); } async function loop() { webcam.update(); await predict(); window.requestAnimationFrame(loop); } async function predict() { const prediction = await model.predict(webcam.canvas); let highProb = 0; let bestName = ""; for (let i = 0; i < maxPredictions; i++) { if (prediction[i].probability > highProb) { highProb = prediction[i].probability; bestName = prediction[i].className; } } const percent = (highProb * 100).toFixed(0); labelContainer.innerHTML = ${bestName}: ${percent}%; // Natijaga qarab rangni o'zgartirish const badge = document.getElementById("result-badge"); if(bestName === "Kitob") badge.style.color = "#818cf8"; else if(bestName === "Kompyuter") badge.style.color = "#c084fc"; else badge.style.color = "#94a3b8"; }
